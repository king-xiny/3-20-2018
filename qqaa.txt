<script>
    var oTab =document.getElementById("box");
    var oUl=oTab.getElementsByTagName("ul")[0];
    var oLis=oUl.getElementsByTagName("li");
    var oDiv=oTab.getElementsByTagName("div");
    var preIndex=0;
    for (var i=0;i<oLis.length;i++){
          oLis[i].temp=i;
          oLis[i].onclick=function () {
              change(this.temp);
          }
    }
    function change(n) {
        oLis[preIndex].className="";
        oDiv[preIndex].className="";
        oLis[n].className="selectLi";
        oDiv[n].className="select";
        preIndex=n;

    }


</script>
<script>
    var obUl=document.getElementById("tag");
    var obLis=obUl.getElementsByTagName("li");
    for(var i=0; i<obLis.length;i++){
        obLis[i].className = (i%2==0)? "pink":"orange";
        obLis[i].temp=obLis[i].className;
        obLis[i].onmouseover=function () {
            this.className="green";
        }
        obLis[i].onmouseout=function () {
            this.className=this.temp;
        }
    }

</script>
<script>
    //    求不确定有效实参个数之和
    function sum() {
        var  total=0;//保存每次累加的值
        for (var i=0;i<arguments.length;i++){
            var cur=arguments[i];
            console.log(cur);
            if(!isNaN(cur)){
                total+= Number(cur);
            }
        }
        return total;
    }
